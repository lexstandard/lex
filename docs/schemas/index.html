Using the Schemas

All schemas target JSON Schema Draft-07 and can be used with any compliant validator.

import jsonschema, json

schema  = json.load(open("schemas/LEX_LEAD_SCHEMA.json"))
message = json.load(open("my-lead.json"))
jsonschema.validate(message["lex"]["payload"], schema)
const Ajv = require("ajv");
const ajv = new Ajv();

const schema  = require("./schemas/LEX_LEAD_SCHEMA.json");
const message = require("./my-lead.json");
const valid   = ajv.validate(schema, message.lex.payload);
SchemaLoader loader = SchemaLoader.builder()
    .schemaJson(new JSONObject(Files.readString(
        Path.of("schemas/LEX_LEAD_SCHEMA.json"))))
    .build();
loader.load().build().validate(payload);
var schema = JsonSchema.FromFile(
    "schemas/LEX_LEAD_SCHEMA.json");
var errors = schema.Validate(jsonPayload);
bool valid  = !errors.Any();
